package ejemplo.trabajo.demo.repository;

import ejemplo.trabajo.demo.model.Libro;
import org.springframework.stereotype.Repository;

import java.util.ArrayList;
import java.util.List;

@Repository
public class LibroRepository {
    //Arreglo que guarda los libros
    private List<Libro> listaLibros = new ArrayList<>();

    //lista que devuelve todos los libros de la lista
    public List<Libro> obtenerLibros(){
        return listaLibros;
    }

    //buscar un libro por la ide 

    public Libro buscarIdLibro(int id){
        for (Libro x : listaLibros){
            if (x.getId() ==  id){
                return x;
            }
        }
        return null;
    }

    //buscar libro por su ISBN

    public Libro buscarPorIsbn(String isbn){
        for(Libro libro : listaLibros){
            if(libro.getIsbn().equals(isbn)){
                return libro;
            }
        }
        return null;
    }

    //metodo guarda el libro en el arreglo y retorna el libo 
    public Libro guardar(Libro lib){
        listaLibros.add(lib);
        return lib;
    }

    //metodo para actualizar el libro por el id
    public Libro actualizar(Libro lib){

        int id = 0;
        int idPosicion = 0;

        for (int i = 0; i < listaLibros.size(); i++){
            if(listaLibros.get(i).getId() == lib.getId()){
                id = lib.getId();
                idPosicion = i;
            }
        }
        Libro libro1 = new Libro();
        libro1.setId(id);
        libro1.setTitulo(lib.getTitulo());
        libro1.setAutor(lib.getAutor());
        libro1.setFechaPublicacion(lib.getFechaPublicacion());
        libro1.setEditorial(lib.getEditorial());
        libro1.setIsbn(lib.getIsbn());

        listaLibros.set(idPosicion, libro1);
        return libro1;
    }

    //metodos utilizados para eliminar items en un arreglo 
    public void eliminar(int id){
        //alternativa 1
        Libro libro = buscarIdLibro(id);
        if (libro != null){
            listaLibros.remove(libro);
        }
    }
}
